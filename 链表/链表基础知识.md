public class ListNode {
   int val;
   
   ListNode next;
   
   public ListNode(){
   
   }
   
   public ListNode(int val){ 
      this.val = val;
   }
   
   public ListNode(int val, ListNode next){
       this.val = val;
       this.next = next;
   }
}

1.双指针法：定义一个空头结点（翻转链表）
ListNode prev = null;
ListNode cur = head;
ListNode tmpNode = null;
while(cur != null){
    //临时指针保存
    tmpNode = cur.next;
    //cur转方向
    cur.next = prev;
    //prev和cur继续往前走
    prev = cur;
    cur = tmpNode;
}
